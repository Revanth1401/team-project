      <input type="text" id="name" placeholder="Full Name" required />
      <input type="password" id="password" placeholder="Password" required />
      <input type="password" id="confirm_password" placeholder="Re-type Password" required />

  <script>
    const password = document.getElementById("password");
    const confirmPassword = document.getElementById("confirm_password");
    const passwordMsg = document.getElementById("passwordMsg");
    const responseMsg = document.getElementById("responseMsg");

    confirmPassword.addEventListener("input", () => {
      if (confirmPassword.value === "") {
        passwordMsg.textContent = "";
        return;
      }

      if (confirmPassword.value === password.value) {
        passwordMsg.textContent = "‚úÖ Passwords match";
        passwordMsg.className = "match-msg match";
      } else {
        passwordMsg.textContent = "‚ùå Passwords do not match";
        passwordMsg.className = "match-msg no-match";
      }
    });

    document.getElementById("registerForm").addEventListener("submit", async function (e) {
      e.preventDefault();

      if (password.value !== confirmPassword.value) {
        alert("Passwords do not match.");
        return;
      }

      const payload = {
        user_id: document.getElementById("user_id").value,
        name: document.getElementById("name").value,
        password: password.value
      };

      try {
        const response = await fetch("/register/", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "X-CSRFToken": getCSRFToken() // If using Django's CSRF middleware
          },
          body: JSON.stringify(payload)
        });

        const data = await response.json();
        if (response.ok) {
          responseMsg.textContent = "üéâ " + data.message;
          responseMsg.className = "success";
          document.getElementById("registerForm").reset();
          passwordMsg.textContent = "";
        } else {
          responseMsg.textContent = "‚ùå " + (data.error || "Registration failed");
          responseMsg.className = "error";
        }
      } catch (err) {
        responseMsg.textContent = "‚ùå Server error.";
        responseMsg.className = "error";
      }
    });

    // Utility: Get CSRF token from cookie
    function getCSRFToken() {
      let name = 'csrftoken=';
      let decodedCookie = decodeURIComponent(document.cookie);
      let ca = decodedCookie.split(';');
      for (let i = 0; i < ca.length; i++) {
        let c = ca[i].trim();
        if (c.indexOf(name) == 0) return c.substring(name.length, c.length);
      }
      return "";
    }
